% ---------- geometry ----------
\def\R{1.0}
\def\dx{2.35}
\def\labelY{-1.45}
\def\amp{0.1}

% ---------- panels 2-5: r(theta) = R + A cos(m theta), transparent fill, no stroke ----------
\foreach \m/\col/\i in {0/ink/0, 1/grayink/1, 2/green/2, 3/magenta/3, 4/green/4, 5/magenta/5}{
  \begin{scope}[shift={(\i*\dx,0)}]
    \setdirection{\i}
    \draw[ink, thin] (0,0) circle (\R);
    \pgfmathsetmacro{\mirror}{or(\m==0,mod(\m,2)==1) ? -1 : 1}
    \pgfmathsetmacro{\opa}{\m==0 ? 0.5 : (\m==1 ? 0.5 : (mod(\m,2)==0 ? 0.85 : 0.75))}
    \pgfmathsetmacro{\ampsign}{\m==0 ? -1 : 1}
    \fill[\col, opacity=\opa]
      plot[domain=0:360, samples=220, smooth cycle]
      ({\mirror*(\R+\ampsign*\amp*cos(\m*\x))*cos(\x)}, {(\R+\ampsign*\amp*cos(\m*\x))*sin(\x)});
    \node at (0,\labelY) {$m=\i$};
  \end{scope}
}

% ---------- trailing ellipsis ----------
\node at (6*\dx-0.15,0) {\Huge $\cdots$};
