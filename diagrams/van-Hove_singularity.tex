\begin{tikzpicture}[scale=1.2, >=stealth]
  % Axes
  \draw[->, thick] (-0.2,0) -- (4.5,0) node[right] {Energy, $E$};
  \draw[->, thick] (0,-0.2) -- (0,3.8) node[above] {Density of States, $g(E)$};

  % Free electron gas background (2D: constant DOS, dashed line)
  \draw[dashed, gray!70, thick] (0, 0) -- (4.2, 0); % Just as a reference
  % Instead of sqrt(E), we use a constant to imply 2D free gas at low energy
  \draw[domain=0.01:4.2, samples=100, smooth, variable=\x, dashed, gray!70, thick]
  plot (\x, {1.2});
  \node[gray!70, right] at (3.5, 1.0) {$g_{2D}(E) \propto \text{const.}$};

  % Real DOS with a highly dramatic 2D-like Divergent van Hove Singularity at E_c=2.2
  % Formula creates a sharp divergence smoothed only by a small epsilon
  \def\ec{2.2}
  \def\epsilon{0.03}
  \draw[domain=0.01:1.6, samples=100, smooth, variable=\x, blue!80!black, ultra thick]
  plot (\x, {1.2 + 0.1*\x}); % Initial slightly-rising, flatish region
  \draw[domain=1.6:\ec-0.005, samples=150, smooth, variable=\x, blue!80!black, ultra thick]
  plot (\x, {1.36 + 1.2 / sqrt((\ec-\x)^2 + \epsilon^2)}); % Left divergence
  \draw[domain=\ec+0.005:4.2, samples=150, smooth, variable=\x, blue!80!black, ultra thick]
  plot (\x, {1.2 + 1.2 / sqrt((\x-\ec)^2 + \epsilon^2)}); % Right divergence

  % Labels and annotations
  % Critical Energy (E_c)
  \draw[red, dashed, thick] (2.2,0) -- (2.2,3.3);
  \fill[red] (2.2,3.3) circle (2pt);
  \node[below] at (2.2,0) {$E_c$};

  % Label for van Hove Singularity text, kept and pointed
  \node[red!90!black, font=\bfseries] at (3.1, 3.5) {van Hove Singularity};
  \draw[<-, thick, red!90!black, >=stealth] (2.2, 3.4) -- (3.1, 3.4);

\end{tikzpicture}
