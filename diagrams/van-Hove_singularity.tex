\begin{tikzpicture}
  \begin{axis}[
      width=6cm, height=5cm,
      xlabel={Energy},
      ylabel={Density of states},
      xmin=-3.15, xmax=3.15,
      ymin=0, ymax=1.2,
      xtick={0},
      ytick={\empty},
      axis lines=box,
      axis line style={black},
      tick style={black, thin},
      tick align=inside,
      every tick/.style={black},
      ticklabel style={font=\small},
      label style={font=\normalsize},
      clip=true,
      samples=250,
      smooth,
    ]
    % --- graphene tight-binding density of states, piecewise to
    % isolate the logarithmic van Hove singularities at E = +-1.
    % Background rises monotonically from E=0 to the singularity at
    % E=1, then decreases monotonically (curving down) all the way
    % to the band edge E=3, where it is cut off vertically. ---
    \def\baseA{0.30}
    \def\edgeK{0.0375}
    \def\vhK{0.24}
    \def\vhW{0.42}
    \pgfmathdeclarefunction{grBase}{1}{%
      \pgfmathparse{ifthenelse(abs(#1)<=1,
        \baseA*abs(#1),
        \baseA - \edgeK*(abs(#1)-1)^2)}%
    }
    \pgfmathdeclarefunction{grDOS}{1}{%
      \pgfmathparse{grBase(#1)
        + \vhK*ln(1/max(abs(abs(#1)-1),0.001))
        *exp(-((abs(#1)-1)/\vhW)^2)}%
    }

    % left band: -3 to just below -1 (spike clips at the top of the box)
    \addplot[thick, blueink, domain=-3:-1.005] {grDOS(x)};
    % middle band: spans across E=0, between the two singularities
    \addplot[thick, blueink, domain=-0.995:0.995] {grDOS(x)};
    % right band: just above +1 to 3
    \addplot[thick, blueink, domain=1.005:3] {grDOS(x)};

    % vertical cutoffs at the band edges E = +-3
    \pgfmathsetmacro{\edgeval}{grDOS(3)}
    \draw[thick, blueink] (axis cs:3,0) -- (axis cs:3,\edgeval);
    \draw[thick, blueink] (axis cs:-3,0) -- (axis cs:-3,\edgeval);
  \end{axis}
\end{tikzpicture}
