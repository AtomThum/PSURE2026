\section{Introduction}

\subsection{Boltzmann equation}

\begin{frame}
  \begin{blk}{Boltzmann equation}
    \begin{equation}
      \left(\pdv{}{t} + \vv{v} \vdot \pdv{}{\vv{r}} + \vv{F} \vdot \pdv{}{\hbar\vv{p}}\right) f(t,\vv{x},\vv{p}) = \hat{\mathcal{J}}[f(t,\vv{x},\vv{p})]
    \end{equation}
  \end{blk}

  This talk follows the treatment of the linearized collision integral in \textcite{hofmann-2025}, which builds on the mode hierarchy first identified in \textcite{mit-excitation-hierarchy}.

  \begin{description}[\(\vv{v} \vdot \pdv{f}{\vv{r}} + \vv{F} \vdot \pdv{f}{\hbar\vv{p}}\):]
    \item[\(f(t, \vv{x}, \vv{p})\):] One-particle density function
    \item[\(\pdv{f}{t}\):] Local time evolution
    \item[\(\vv{v} \vdot \pdv{f}{\vv{r}}\):] Streaming/advection term
    \item[\(\vv{F} \vdot \pdv{f}{\hbar\vv{p}}\):] Force/acceleration term due to external fields
    \item[\(\hat{\mathcal{J}}\lbrack f \rbrack\):] Collision integral - effect of collisions on distribution function
  \end{description}
\end{frame}

\subsection{Fermi Dirac distribution}

\begin{frame}
  \begin{blk}{Fermi-Dirac distribution}
    \begin{equation}
      f_0(\vv{p}) = \frac{1}{\exp[\beta(\epsilon(\vv{p}) - \mu)] + 1}
    \end{equation}
  \end{blk}

  \begin{itemize}
    \item \(\beta = 1/k_B T\) is the inverse temperature
    \item \(\epsilon(\vv{p}) = \frac{\hbar^2 \vv{p}^2}{2m}\) is the energy of particles with momentum \(\vv{p}\)
    \item \(\mu\) is the chemical potential
  \end{itemize}

  The Fermi-Dirac distribution is in \alert{equilibrium}.
\end{frame}

\begin{frame}
  \begin{figure}[]
    \centering
    \includegraphics[scale=0.75]{fermi-dirac_distribution.pdf}
    \caption{Fermi-Dirac distribution at different temperatures}
  \end{figure}
  \vspace{-1.5em}

  \begin{itemize}
    \item At \(T = 0\), all states below the Fermi energy are filled, all states above are empty.
    \item As \(T\) increases, the distribution smooths out.
  \end{itemize}
\end{frame}

\begin{frame}{\(k\)-space}
  \begin{itemize}
    \item Analogous to momentum space
    \item Each point \(\vv{k}\) in \(k\)-space corresponds to a wave $\e^{\im \vv{k}}$
    \item \alert{Fourier transform} converts between \(k\)-space to real space
  \end{itemize}
\end{frame}

\begin{frame}
  \begin{figure}[]
    \centering
    \includegraphics[scale=0.75]{2D_k-space.pdf}
    \caption{Representative waves of \(k\)-space}
  \end{figure}

  \vspace{-1.5em}
  \begin{itemize}
    \item Distance from origin \(\propto\) Wave frequency
    \item Direction on \(k\)-space \(\implies\) Wave direction
  \end{itemize}
\end{frame}

\begin{frame}{Fermi surface}
  \begin{itemize}
    \item Surface in \(k\)-space that separates \emph{occupied electron states} from \emph{unoccupied} electron states
    \item Determined by periodicity and symmetry of crystal lattice.
    \item Changes with temperature.
  \end{itemize}
\end{frame}

\begin{frame}{Collision integral}
  \begin{blk}{}
    \vspace{-\abovedisplayskip}
    \begin{gather}
      \hat{\mathcal{J}}[f] = - \dotsint \frac{1}{(2\cpi)^6} \odif{\vv{p}_2,\vv{p}_1',\vv{p_2}'} W(\vv{p}_1', \vv{p}_2' | \vv{p}_1, \vv{p}_2) \mathcal{B}, \\
      \mathcal{B} = f_1 f_2 (1 - f_{1'})(1 - f_{2'}) - f_{1'} f_{2'} (1 - f_1)(1 - f_2), \\
      f_i = f(\vv{p}_i), \quad f_{0, i} = f_0(\vv{p}_i).
    \end{gather}
  \end{blk}

  \begin{description}[\(W(\vv{p}_1', \vv{p}_2' | \vv{p}_1, \vv{p}_2):\)]
    \item [\(W(\vv{p}_1', \vv{p}_2' | \vv{p}_1, \vv{p}_2):\)] Collision rates
    \item [\(\mathcal{B}:\)] "Product of Fermi factors"\footnote{Non-standard in literatures}
  \end{description}
\end{frame}

\begin{frame}
  \begin{equation}
    \left(\pdv{}{t} + \vv{v} \vdot \pdv{}{\vv{r}} + \vv{F} \vdot \pdv{}{\hbar\vv{p}}\right) f(t,\vv{x},\vv{p}) = - \dotsint \frac{1}{(2\cpi)^6} \odif{\vv{p}_2,\vv{p}_1',\vv{p_2}'} W(\vv{p}_1', \vv{p}_2' | \vv{p}_1, \vv{p}_2) \mathcal{B}
  \end{equation}

  \begin{itemize}
    \item Very hard (might be impossible) to solve analytically
    \item Non-linear
  \end{itemize}
\end{frame}

\begin{frame}{Ideas of attacking the equation}
  \begin{itemize}
    \item Consider only perturbations near equilibrium: \(f(t, \vv{p}) = f_0(\vv{p}) + \fdif{f}(t, \vv{p})\)
    \item Linearize \(\hat{\mathcal{J}}[f]\)
    \item Let there be no external force and no thermal gradient
  \end{itemize}

  We will end up with
  \begin{equation}
    \hat{\mathcal{J}}[\phi] = -\gamma\phi
  \end{equation}
\end{frame}

\begin{frame}
  \begin{figure}
    \centering
    \includegraphics[scale=0.75]{fermi-surface_perturbations.pdf}
    \caption{Circular harmonical perturbations of the Fermi surface}
  \end{figure}

  \begin{enumerate}
    \item Perturbation is smooth
    \item Collision integral is \emph{linear}
  \end{enumerate}
  The perturbation can be expanded in the harmonics basis: \(\e^{\im m \theta}\).
\end{frame}
