\section{Further works}

\begin{frame}{van Hove singularity}
  \begin{blk}{van Hove singularity:}
    An \alert{energy level} that has infinitely many available states.\footnote{Spike in the density of states}
  \end{blk}

  \begin{itemize}
    \item More available states \(\implies\) stronger pairing interactions (forming Cooper pairs)
    \item Superconductivity can happen
  \end{itemize}
\end{frame}

\begin{frame}
  If Fermi surface has the \alert{energy that's close to the van Hove singularity}, superconductivity can occur more easily.

  \alert{Questions:}
  \begin{itemize}

    \item What's the relaxation time?
  \end{itemize}
\end{frame}
