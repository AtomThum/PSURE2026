\section{Odd-Even effect}

\begin{frame}{Trivial solution}
  \(\gamma_m = 0\) is associated with \alert{conserved quantities}.

  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item \(\gamma_0 = 0\)
              \begin{enumerate}
                \item \(\phi_0 = 1\): particle number
                \item \(\phi_0 = \epsilon(\vv{p})\): energy
              \end{enumerate}
      \end{itemize}
    \end{column}
    \begin{column}{0.51\textwidth}
      \begin{itemize}
        \item \(\gamma_1 = 0\)
              \begin{enumerate}
                \item \(\phi_1 = \e^{\im \theta}\): \(x\)-momentum
                \item \(\phi_1 = \e^{-\im \theta}\): \(y\)-momentum
              \end{enumerate}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}
  \begin{columns}
    \begin{column}{0.6\textwidth}
      Near the Fermi surface,
      \begin{equation}
        \epsilon(\vv{p}_1) \approx \epsilon(\vv{p}_2) \approx \epsilon(\vv{p}_1') \approx \epsilon(\vv{p}_2') \approx \epsilon_F.
      \end{equation}
      Thus, \alert{conservation of energy} implies \alert{conservation of momentum magnitude}:
      \begin{equation}
        p_1 + p_2 = p_1' + p_2'.
      \end{equation}
      Three types of collisions that satisfy this constraint:
      \begin{itemize}
        \item Head-on collisions: \(\vv{p}_1 = -\vv{p}_2, \vv{p}_1' = -\vv{p}_2'\)
        \item Forward exchange: \(\vv{p}_1' = \vv{p}_2, \vv{p}_2' = \vv{p}_1\)
        \item No collision: \(\vv{p}_1' = \vv{p}_1, \vv{p}_2' = \vv{p}_2\)
      \end{itemize}
    \end{column}
    \begin{column}{0.4\textwidth}
      \begin{figure}[]
        \centering
        \includegraphics{collision_types.pdf}
        \caption{Relevant collision types}
      \end{figure}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}
  \begin{description}
    \item [\textbf{No collisions:}] Has no contribution to the collision integral.
    \item [\textbf{Forward exchange \& No collisions:}]
          \begin{itemize}
            \item \(\vv{p}_1 = \vv{p}_1'\), \(\vv{p}_2 = \vv{p}_2'\).
            \item \(\psi(\vv{p}_1) + \psi(\vv{p}_2) - \psi(\vv{p}_1') - \psi(\vv{p}_2') = 0\)\footnote{Recall that this is a multiplier in the collision integral}
            \item \(\therefore\) These collisions has no contribution to the collision integral.
          \end{itemize}
  \end{description}

  Only \alert{head-on collision} contribute to the collision integral.
\end{frame}

\begin{frame}{Head on collision}
  \begin{columns}
    \begin{column}{0.4\textwidth}
      \begin{figure}
        \centering
        \includegraphics{head-on_collision.pdf}
        \caption{Head-on collision angle}
      \end{figure}
    \end{column}
    \begin{column}{0.6\textwidth}
      \begin{equation}
        \theta_2 = \theta_1 + \cpi, \mathand \theta_2' = \theta_1' + \cpi.
      \end{equation}
      Substituting test mode \(m\), and recalling \(\e^{\im m (\theta + \cpi)} = (-1)^m \e^{\im m \theta}\), then
      \begin{multline}
        \psi(\vv{p}_1) + \psi(\vv{p}_2) - \psi(\vv{p}_1') - \psi(\vv{p}_2') \\
        = (1 + (-1)^m)\e^{\im m \theta_1} - (1 + (-1)^m)\e^{\im m \theta_{1'}}.
      \end{multline}
      \vspace{-1.5em}
      \begin{description}
        \item [\(m\) is odd:] the term vanishes; thus the collision integral vanishes.
        \item [\(m\) is even:] the term does not vanish.
      \end{description}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{Overview}}
  \begin{itemize}
    \item Use \alert{auxiliary variable} \(\delta_1 = X-\xi\vv{P}\vdot\vv{Q}\) and \(\delta_2 = X+\xi\vv{P}\vdot\vv{Q}\), together with \(\delta_1=0\), and \(\theta_{12} = \theta_2 - \theta_1\).
    \item Collision integral over \(\vv{p}_2\) and \(\vv{\Omega}\). Decompose \(\vv{p}_2\) into \alert{magnitude} \(p_2\) and \alert{angle} \(\theta_2\).
    \item \alert{Change integration variable} from \(p_2\) to \(\delta_2\) and \(\theta_2\) to \(\theta_{12}=\theta_2-\theta_1\).
    \item \alert{Omega integral}: Approximate far from \(\theta_{12}=0,\pi\), reduces to standard hyperbolic integral. Gains a \(T\) factor.
    \item \alert{\(\delta_2\) integral}: Standard numeric integral. Gains another \(T\) factor from change of variables.
    \item \alert{\(\theta_{12}\) integral}: Seemingly diverges, but cut off by \(\Omega\) integral. Gains a \(\log(T)\) factor.
  \end{itemize}
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{Auxiliary Variables}}
  Recall \(X-\xi\vv{P}\vdot\vv{Q} = \beta(\epsilon(\vv{p}_1)-\mu), X+\xi\vv{P}\vdot\vv{Q} = \beta(\epsilon(\vv{p}_2)-\mu)\). Define
  \begin{align}
             \delta_1 = X-\xi\vv{P}\vdot\vv{Q}, &\mathand \delta_2 = X+\xi\vv{P}\vdot\vv{Q}. \\
    \implies X=\frac{\delta_1+\delta_2}{2},     &\mathand \xi\vv{P}\vdot\vv{Q} = \frac{\delta_2-\delta_1}{2}
  \end{align}
  Assuming \(\epsilon(\vv{p}_1) \approx \mu\), \(\delta_1 \approx 0\), so \(X \approx \xi\vv{P}\vdot\vv{Q} \approx \frac{\delta_2}{2}\). 
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{Auxiliary Variables}}
  Define \(\theta_{12} = \theta_2 - \theta_1\). Then
  \begin{align}
    & P=2p_F\ab|\cos(\theta_{12}/2)|, \mathand Q=p_F\ab|\sin(\theta_{12}/2)|. \\
    & \implies \xi\vv{P}\vdot\vv{Q'} = \xi PQ \sin\Omega = \xi p_F^2|\sin(\theta_{12})|\sin\Omega. \\
  \end{align}
  Let \(\Lambda = \epsilon_F \beta = \xi p_F^2\). Finally, \(\xi\vv{P}\vdot\vv{Q'}=\Lambda|\sin(\theta_{12})|\sin\Omega\).
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{Integration Variables}}
  The collision integral has the form (note the first \((\dots)\) contains constants, the second contains dimensionless inteegration quantities like \(\phi\)).
  \begin{equation}
    (\dots) \dotsint \odif{\vv{p}_2} \odif{\Omega} F_{121'2'} (\dots)
  \end{equation}
  Using \(\odif{\vv{p}_2} = p_2\odif{p_2}\odif{\theta_2}, \odif{\theta_2} = \odif{\theta_{12}}, \mathrm{and} \odif{\delta_2} = \beta\odif{\epsilon(p_2)}=\beta\frac{\hbar^2p_2}{m^*}\), we obtain the form
  \begin{equation}
    (\dots) \dotsint \odif{\delta_2} \odif{\theta_{12}} \odif{\Omega} F_{121'2'} (\dots)
  \end{equation}
  with constants absorbed into the first \((\dots)\) (restored later on). Recall $\beta=\frac{1}{k_B T}$. \alert{This is our first $T$ factor}.
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{\(F_{121'2'}\)}}
  \begin{align}
    F_{121'2'} &= \frac{1}{4[\cosh{X}+\cosh{\xi\vv{P}\vdot\vv{Q}}][\cosh{X}+\cosh{\xi\vv{P}\vdot\vv{Q'}}]} \\
               &= \frac{1}{4[\cosh{(\delta_1+\delta_2)/2}+\cosh{(\delta_2-\delta_1)/2}][\cosh{(\delta_1+\delta_2)/2}+\cosh{(\Lambda|\sin(\theta_{12})|\sin\Omega})]} \\
               &= \frac{1}{4[\cosh{\delta_2/2}+\cosh{\delta_2/2}][\cosh{\delta_2/2}+\cosh{(\Lambda|\sin(\theta_{12})|\sin\Omega})]} \\
               &= \frac{1}{8\cosh{\delta_2/2}[\cosh{\delta_2/2}+\cosh{(\Lambda|\sin(\theta_{12})|\sin\Omega})]}.
  \end{align}
  where we have used \(\delta_1=0\). 
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{Omega Integral}}
  Collecting terms that depend on \(\Omega\), we have
  \begin{equation}
    \int_0^{2\pi} \frac{\odif{\Omega}}{\cosh{\delta_2/2}+\cosh{(\Lambda|\sin(\theta_{12})|\sin\Omega})}.
  \end{equation}
  For \(\theta_{12}\) not near \(0\) or \(\pi\), approximate \(\sin\Omega \approx u\), extend the integral to \((-\infty, \infty)\), and change integration variables to \(v=\Lambda|\sin(\theta_{12})|u\). Then the integral becomes
  \begin{align}
    &= \int_{-\infty}^{\infty} \frac{\odif{\Omega}}{\cosh{\delta_2/2}+\cosh{(\Lambda|\sin(\theta_{12})|\Omega})} 
    = \frac{1}{\Lambda|\sin(\theta_{12})|} \int_{-\infty}^{\infty}  \frac{\odif{v}}{\cosh{\delta_2/2}+\cosh{v}} \\
    &= \frac{2\delta_2}{\Lambda|\sin(\theta_{12})|\sinh{\delta_2/2}}
  \end{align}
  Where we have used the standard integral \(\int_{-\infty}^{\infty} \frac{\odif{x}}{\cosh{a}+\cosh{x}} = \frac{2a}{\sinh{a}}\).
\end{frame}

\begin{frame}{Finding the Even Decay Rate : \alert{\(\delta_{12}\) Integral}}
  Collecting terms that depend on \(\delta_2\), we have
  \begin{equation}
    \frac{1}{8\cosh{\delta_2/2}} \frac{2\delta_2}{\Lambda\sin(\theta_{12})|\sinh{\delta_2/2}} = \frac{\delta_2}{2\Lambda\sin(\theta_{12})|\sinh{\delta_2}}.
  \end{equation}
  Using the standard integral \(\int_{-\infty}^{\infty} \frac{x}{\sinh{x}} \odif{x} = \frac{\pi^2}{2}\), we have
  \begin{equation}
    \int_{-\infty}^{\infty} \frac{\delta_2}{2\Lambda\sin(\theta_{12})|\sinh{\delta_2}} \odif{\delta_2} = \frac{\pi^2}{4\Lambda|\sin(\theta_{12})|}.
  \end{equation}
\end{frame}
